import 'package:flutter/material.dart';
import 'package:shop/core/app_colors.dart';
import 'package:shop/screen/home_screen/taps/cart/cart.dart';
import 'package:shop/screen/home_screen/taps/maps/map.dart';
import 'package:shop/screen/home_screen/taps/home/home.dart';
import 'package:shop/screen/home_screen/taps/profile/profile.dart';

class HomeScreen extends StatefulWidget {
  const HomeScreen({super.key});

  @override
  State<HomeScreen> createState() => _HomeScreenState();
}

class _HomeScreenState extends State<HomeScreen> {
  int navIndex = 0;

  List<Widget> pages = const [Home_tap(), Search(), Browse(), Profile()];
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Row(
          spacing: 10,
          children: [
            Image.asset("assets/images/logo.png", height: 40),
            Text("Shop"),
          ],
        ),
      ), 
      body: pages[navIndex],
     bottomNavigationBar: BottomNavigationBar(
        currentIndex: currentNavIndex,
        showSelectedLabels: true,
        elevation: 0,
        onTap: (index) {
          setState(() {
            currentNavIndex = index;
          });
        },
        items: [
          BottomNavigationBarItem(
            icon: SvgPicture.asset(AppImages.homeIcon),
            label: l10n.home,
            activeIcon: SvgPicture.asset(AppImages.homeFill),
          ),
          BottomNavigationBarItem(
            icon: SvgPicture.asset(AppImages.location),
            label: l10n.map,
            activeIcon: SvgPicture.asset(AppImages.mapFill),
          ),
          BottomNavigationBarItem(
            icon: SvgPicture.asset(AppImages.heart),
            label: l10n.favorite,
            activeIcon: SvgPicture.asset(AppImages.loveFill),
          ),
          BottomNavigationBarItem(
            icon: SvgPicture.asset(AppImages.person),
            label: l10n.profile,
            activeIcon: SvgPicture.asset(AppImages.personFil),
          ),
        ],
      ),
    );
  }
}
